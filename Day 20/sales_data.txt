Step 2 – Read CSV
import pandas as pd

df = pd.read_csv("sales_data.csv")

print(df)
🎯 Task 1

Basic Information

print(df.head())

print(df.info())

print(df.describe())
🎯 Task 2

Missing Values

print(df.isnull().sum())

Fill Missing Sales

df["Sales"] = df["Sales"].fillna(df["Sales"].mean())
🎯 Task 3

Convert Date

df["Date"] = pd.to_datetime(df["Date"])

New Columns

df["Month"] = df["Date"].dt.month_name()

df["Year"] = df["Date"].dt.year
🎯 Task 4

Highest Sale

print(df["Sales"].max())

Lowest Sale

print(df["Sales"].min())

Average Sale

print(df["Sales"].mean())
🎯 Task 5

Filter

Laptop Sales

print(df[df["Category"] == "Laptop"])

Delhi Sales

print(df[df["City"] == "Delhi"])
🎯 Task 6

GroupBy

City Wise Sales

print(
    df.groupby("City")["Sales"].sum()
)

Category Wise Sales

print(
    df.groupby("Category")["Sales"].mean()
)
🎯 Task 7

Pivot Table

print(
    pd.pivot_table(
        df,
        values="Sales",
        index="City",
        columns="Category",
        aggfunc="sum",
        margins=True
    )
)
🎯 Task 8

Performance Column

df["Performance"] = df["Sales"].apply(
    lambda x: "High"
    if x >= 40000
    else "Low"
)
🎯 Task 9

Running Total

df["Running_Total"] = df["Sales"].cumsum()

rolling average

df["Rolling_Avg"] = df["Sales"].rolling(3).mean()
🎯 Task 10

Final Report

print(df)
⭐ Bonus Challenge 1

Create Grade

def grade(x):
    if x >= 50000:
        return "A"

    elif x >= 40000:
        return "B"

    else:
        return "C"

df["Grade"] = df["Sales"].apply(grade)
⭐ Bonus Challenge 2

Top 3 Sales

print(
    df.nlargest(
        3,
        "Sales"
    )
)

Bottom 3

print(
    df.nsmallest(
        3,
        "Sales"
    )
)
⭐ Bonus Challenge 3

Monthly Sales

print(
    df.groupby("Month")["Sales"].sum()
)
